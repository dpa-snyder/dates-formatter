# Based on _detect_data_kind logic (GUI line 1531).

def detect_date_columns(df, threshold=0.6):
    """
    Scan each column. Return list of names where >= threshold of
    non-empty values match a known date pattern.
    Samples up to 200 rows per column for speed.
    """
    date_cols = []
    patterns = [
        re.compile(r'^\d{1,2}/\d{1,2}/\d{4}'),
        re.compile(r'^\d{4}-\d{2}-\d{2}'),
        re.compile(r'^\d{4}$'),
        re.compile(r'^\d{4}-\d{4}$'),
        re.compile(r'^\d{4}-\d{2}$'),
        re.compile(r'(?i)\b(circa|ca\.|before|after|undated)\b'),
    ]
    for col in df.columns:
        vals = [str(v).strip() for v in df[col].dropna() if str(v).strip()][:200]
        if not vals:
            continue
        hits = sum(1 for v in vals if any(p.search(v) for p in patterns))
        if hits / len(vals) >= threshold:
            date_cols.append(col)
    return date_cols

# web/app.py
import io, json, os, tempfile, uuid
from flask import Flask, request, jsonify, send_file, send_from_directory
import pandas as pd
import sys

sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..', 'prod'))
import date_engine

app = Flask(__name__)
TOKENS       = {}   # {uuid: {'path': str, 'df': DataFrame, 'ext': str}}
PROD_DIR     = os.path.join(os.path.dirname(__file__), '..', 'prod')
SETTINGS_PATH = os.path.join(PROD_DIR, 'dates-formatter-settings.json')
DEFAULT_SETTINGS = {'theme': 'dark', 'output_mode': 'overwrite', 'recent_files': []}
VALID_KEYS   = set(DEFAULT_SETTINGS)

@app.route('/')
def index():
    return send_from_directory(
        os.path.join(os.path.dirname(__file__), 'templates'), 'index.html')

@app.route('/api/load', methods=['POST'])
def load_file():
    f = request.files.get('file')
    if not f or not f.filename:
        return jsonify(error='No file provided.'), 400
    ext = os.path.splitext(f.filename)[1].lower()
    if ext not in ('.xlsx', '.csv'):
        return jsonify(error='Unsupported file type. Use .xlsx or .csv.'), 400
    tmp_dir  = tempfile.mkdtemp(prefix='df-')
    tmp_path = os.path.join(tmp_dir, f.filename)
    f.save(tmp_path)
    try:
        df = pd.read_excel(tmp_path) if ext == '.xlsx' else pd.read_csv(tmp_path)
    except Exception as e:
        return jsonify(error=f'Could not read file: {e}'), 400
    token = str(uuid.uuid4())
    TOKENS[token] = {'path': tmp_path, 'df': df, 'ext': ext}
    return jsonify(token=token, columns=list(df.columns),
                   row_count=len(df),
                   date_columns=date_engine.detect_date_columns(df))

@app.route('/api/process', methods=['POST'])
def process():
    data   = request.get_json(force=True) or {}
    token  = data.get('token')
    column = data.get('column')
    mode   = data.get('mode')
    if not all([token, column, mode]):
        return jsonify(error='token, column, and mode are required.'), 400
    if token not in TOKENS:
        return jsonify(error='Token not found or expired.'), 404
    if mode not in ('single', 'ae', 'dublin'):
        return jsonify(error=f'Unknown mode: {mode}'), 400
    entry = TOKENS.pop(token)
    try:
        df, flagged = date_engine.process_column(entry['df'].copy(), column, mode)
    except Exception as e:
        return jsonify(error=str(e)), 500
    buf = io.BytesIO()
    ext = entry['ext']
    if ext == '.xlsx':
        with pd.ExcelWriter(buf, engine='openpyxl') as w:
            df.to_excel(w, index=False)
            ws = w.sheets[next(iter(w.sheets))]
            for row in ws.iter_rows(min_row=2):
                for cell in row:
                    cell.number_format = '@'  # CRITICAL — prevent Excel auto-conversion
        mime, fname = ('application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',
                       'output.xlsx')
    else:
        buf.write(df.to_csv(index=False).encode())
        mime, fname = 'text/csv', 'output.csv'
    buf.seek(0)
    resp = send_file(buf, mimetype=mime, as_attachment=True, download_name=fname)
    resp.headers['X-Flagged-Count'] = str(flagged)
    resp.headers['X-Row-Count']     = str(len(df))
    return resp

@app.route('/api/settings', methods=['GET', 'POST'])
def settings():
    def _read():
        try:
            with open(SETTINGS_PATH, 'r', encoding='utf-8') as fh:
                return {**DEFAULT_SETTINGS, **json.load(fh)}
        except (OSError, json.JSONDecodeError):
            return dict(DEFAULT_SETTINGS)
    if request.method == 'GET':
        return jsonify(_read())
    incoming = request.get_json(force=True) or {}
    bad = [k for k in incoming if k not in VALID_KEYS]
    if bad:
        return jsonify(error=f'Unknown settings key(s): {bad}'), 400
    merged = {**_read(), **{k: v for k, v in incoming.items() if k in VALID_KEYS}}
    with open(SETTINGS_PATH, 'w', encoding='utf-8') as fh:
        json.dump(merged, fh, indent=2)
    return jsonify(merged)

# prod/launch.py
import os, socket, sys, threading, time, webbrowser

def find_free_port():
    with socket.socket() as s:
        s.bind(('', 0))
        return s.getsockname()[1]

def wait_for_server(port, timeout=8.0):
    import urllib.request
    deadline = time.time() + timeout
    while time.time() < deadline:
        try:
            urllib.request.urlopen(f'http://127.0.0.1:{port}/', timeout=1)
            return True
        except Exception:
            time.sleep(0.15)
    return False

if __name__ == '__main__':
    # Resolve web/ directory relative to this script
    here = os.path.dirname(os.path.abspath(__file__))
    sys.path.insert(0, os.path.join(here, '..', 'web'))
    from app import app, _shutdown

    port = find_free_port()
    t = threading.Thread(
        target=app.run,
        kwargs={'host': '127.0.0.1', 'port': port, 'debug': False},
        daemon=True)
    t.start()
    print(f'Dates Formatter starting on http://127.0.0.1:{port}')
    if not wait_for_server(port):
        print('Server did not start. Is flask installed?')
        sys.exit(1)
    webbrowser.open(f'http://127.0.0.1:{port}')
    print('Press Ctrl+C to quit.')
    try:
        while not _shutdown:
            time.sleep(0.5)
    except KeyboardInterrupt:
        pass
    print('Shutting down.')

Critical Pitfalls

Read Before Coding ▾
  • #1
    custom_format_date() returns a TUPLE (str, flag) — not a string.
    Flag is 'Yes' or ''. The range pipeline unpacks it: v, f = custom_format_date(val). If you call it and assign to a string variable, you get a tuple in the column and the entire output is silently wrong. Check all call sites.
  • #2
    date_engine.py must have zero side effects on import.
    The GUI has SETTINGS = load_settings() at line 94 (module level). Do NOT copy this into the engine. The engine must be importable with no file I/O, no logging setup, no global state mutation. Flask imports it at startup and imports must be fast and clean.
  • #3
    Test suite imports from src/date-formatter-gui.py via importlib.
    After W-103 the GUI file uses from date_engine import .... The imported names must be accessible at module level in the GUI (they are, via Python's normal import re-export). Verify with ./run-tests.sh before moving to Phase 2. If tests fail after W-103, the fix is adding explicit re-exports in the GUI file.
  • #4
    Excel number_format = '@' is non-negotiable.
    Without this, Excel interprets MM/DD/YYYY strings as dates and auto-converts them to serial numbers or its own display format on open. Apply to every data cell (rows 2+, all columns) in the ExcelWriter block. This is done in the existing CTk save path and must be done identically in /api/process.
  • #5
    apply_leading_zeros(df) runs for every mode, every column — never skip it.
    It pads RG columns to 4 chars and SG/SubGr/Series etc. to 3 chars. It only touches columns that exist in the dataframe — safe to always call. Lives at the end of every pipeline branch in _process_column.
  • #6
    reorder_columns must run AFTER all processing.
    It moves Original_{col} and Check {col} to immediately follow the source column. If called before the pipeline writes to those columns, the column order is wrong. Always call at end, before apply_leading_zeros.
  • #7
    Flask file download: use io.BytesIO, call buf.seek(0) before send_file.
    For xlsx: the pd.ExcelWriter context must be closed (exit the with block) before buf.seek(0). If you seek before the context closes, the buffer is empty. Order: open BytesIO → enter ExcelWriter with buf → write → exit with block → seek(0) → send_file.
  • #8
    Windows PermissionError on save = file open in Excel.
    The CTk GUI has a retry dialog (messagebox.askretrycancel). Flask cannot show a dialog. Instead return 409 Conflict with {"error": "Output file is locked. Close it in Excel and retry."}. The frontend shows this inline with a retry button that re-calls /api/process with the same token — but the token was consumed. Solution: either don't consume token on PermissionError, or have the frontend re-upload. Simplest: catch PermissionError before popping the token, return 409 with token still intact.
  • #9
    output_mode: 'copy' doesn't map cleanly to download flow.
    In the CTk GUI, 'copy' saves to {name}_formatted{ext} alongside the source. In the web flow, Flask sends a download and the user picks the save location — there is no "alongside" concept. Implement 'copy' as: download filename becomes output_formatted.xlsx vs output.xlsx. Document this difference in the UI.
  • #10
    Do not put web/ inside prod/.
    web/ lives at repo root, alongside prod/. launch.py in prod/ adjusts sys.path to find web/app.py and prod/date_engine.py. If you put web/ inside prod/, the send_from_directory paths and sys.path adjustments in the stubs will need to change.
